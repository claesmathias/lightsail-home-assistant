#!/bin/bash
# install the necessary tools
sudo yum install docker git epel-release -y

# add new group and add centos user
sudo groupadd docker
sudo usermod -aG docker centos

# enable docker at startup
sudo systemctl enable docker.service
sudo systemctl start docker.service

# install docker-compose
sudo curl -L "https://github.com/docker/compose/releases/download/1.24.0/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

# clone
cd /home/centos/
git clone https://github.com/claesmathias/lightsail-home-assistant.git

# set docker-compose environment variables
cat <<EOF >/home/centos/lightsail-home-assistant/.env
HOST_IP=$(hostname --ip-address)
EMAIL=mail@example.com
URL=example.com
SUBDOMAINS=www
EXTRA_DOMAINS=
DUCKDNS_SUBDOMAINS=subdomain.duckdns.org
DUCKDNS_TOKEN=duck-dns-token
TZ=Europe/Brussels
EOF

# set permissions
sudo chown -R centos:centos /home/centos/lightsail-home-assistant/

# set selinux permissions
sudo chcon -Rt svirt_sandbox_file_t /home/centos

# start docker containers
cd /home/centos/lightsail-home-assistant/
/usr/local/bin/docker-compose up -d

# reset ssh permissions
restorecon -FRvv ~/.ssh
